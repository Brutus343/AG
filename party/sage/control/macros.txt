automacro checkDexLower {
   location mosk_dun01
    timeout 30
	disabled 1
    call {
        # Calculate total DEX (Base + Bonus)
        $totalDex = @eval($::char->{int} + $::char->{int_bonus})

        # Check if the total is less than 50
        if ($totalDex < 98) {
            log Total DEX is $totalDex. Sending alert!
			do move stop
			pause 0.5
            do p lowstats
			pause 5
        }
    }
}
