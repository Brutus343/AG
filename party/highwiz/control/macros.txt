automacro stuck {
console /Stuck during route/
call {
do c @refresh
}
timeout 5
}

automacro castSwarmStormGust {
    party /Multiple monsters engaged/
    exclusive 1
    timeout 3
    call {
        log Emergency Storm Gust on SilverShield!
        sp 89 SilverShield
    }
}
